% test_item_table.tex
%
% Defines a reusable command for creating a table description of a test item.
%
% This command, \testitemtable, takes one argument which is the body of the table.
% The table body should be a series of rows, with each row ending in \\ \hline.
%
% Example usage:
% \testitemtable{
% Name/Code & Serial No./Revision & Manufacturer & Configuration & Images \\
% \hline
% }
\newcommand{\testitemtable}[1]{
    \begin{tabular}{|p{2.5cm}|p{3.5cm}|p{2.4cm}|p{2.4cm}|p{2.4cm}|}
    \hline
    \textbf{Name/Code} & \textbf{Serial No./Revision} & \textbf{Manufacturer} & \textbf{Configuration} & \textbf{Images} \\
    \hline
    #1
    \end{tabular}
}
